import java.util.Scanner; import java.util.Random; public class Main { public static void main(String[] args) { Scanner scanner = new Scanner(System.in); Random random = new Random(); System.out.println("Projekt: Schere Stein Papier"); while (true) { System.out.println("Wähle: 0 = Schere, 1 = Stein, 2 = Papier, 3 = Beenden"); if (!scanner.hasNextInt()) { scanner.next(); continue; } int spielerWahl = scanner.nextInt(); if (spielerWahl == 3) { System.out.println("Das Spiel wurde beendet."); break; } if (spielerWahl < 0 || spielerWahl > 2) { System.out.println("Ungültige Spielereingabe"); continue; } int computerWahl = random.nextInt(3); System.out.println("Deine Wahl: " + convert(spielerWahl)); System.out.println("Wahl des Computers: " + convert(computerWahl)); if (spielerWahl == computerWahl) { System.out.println("Unentschieden."); } else if ((spielerWahl == 0 && computerWahl == 2) || (spielerWahl == 1 && computerWahl == 0) || (spielerWahl == 2 && computerWahl == 1)) { System.out.println("Du hast gewonnen!"); } else { System.out.println("Der Computer hat gewonnen"); } System.out.println(); } scanner.close(); } public static String convert(int wahl) { if (wahl == 0) return "Schere"; if (wahl == 1) return "Stein"; return "Papier"; } }
